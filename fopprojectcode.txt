#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 6
#define MINES 15

char board[SIZE][SIZE]; // actual mine board
char visible[SIZE][SIZE]; // what tthe  player  sees

void initBoards() {
  for (int i = 0; i < SIZE; i++) {
    for (int j = 0; j < SIZE; j++) {
      board[i][j] = '.';
      visible[i][j] = '#';
    }
  }
}

void placeMines() {
  int placed = 0;

  while (placed < MINES) {
    int r = rand() % SIZE;
    int c = rand() % SIZE;

    if (board[r][c] != '^') {
      board[r][c] = '^';
      placed++;
    }
  }
}

int countMines(int row, int col) {
  int count = 0;

  for (int i = -1; i <= 1; i++) {
    for (int j = -1; j <= 1; j++) {
      int r = row + i;
      int c = col + j;

      if (r >= 0 && r < SIZE && c >= 0 && c < SIZE) {
        if (board[r][c] == '*')
          count++;
      }
    }
  }

  return count;
}

void printBoard() {
  printf("\n");

  for (int i = 0; i < SIZE; i++) {
    for (int j = 0; j < SIZE; j++) {
      printf("%c ", visible[i][j]);
    }
    printf("\n");
  }

  printf("\n");
}
